to fern :size :sign
  if :size < 1 [ stop ] fd :size rt 70 * :sign fern :size * 0.5 :sign * -1 lt 70 * :sign fd :size lt 70 * :sign fern :size * 0.5 :sign rt 70 * :sign rt 7 * :sign fern :size - 1 :sign lt 7 * :sign bk :size * 2
end

to randomcolor
  setcolor pick [ red orange yellow green blue violet ]
end

to square :length
  repeat 4 [ fd :length rt 90 ]
end

to star
  repeat 5 [ fd 100 rt 144 ]
end

to tree :size
  if :size < 5 [ forward :size back :size stop ] forward :size / 3 left 30 tree :size * 2 / 3 right 30 forward :size / 6 right 25 tree :size / 2 left 25 forward :size / 3 right 25 tree :size / 2 left 25 forward :size / 6 back :size
end

to pave

end

to fw
  penup fd 10 wait 60
end

to tr
